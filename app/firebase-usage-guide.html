Set fireBase url in angular.config intial setup of firebase service
            
                //Adding firebase module as dependency for angularjs app
                var myapp = app.module('angularappname',['firebase-db-module',....])
                ...

                //injecting firebase provider to set primary settings like url.,
                myapp.config(function(firebasedbProvider,...){


                //setting firebase server url as configurations for firebase module
                var firebaseConfig = {};
                firebaseConfig.Url = 'https://nimbu-polling.firebaseio.com';

                firebasedbProvider.config(firebaseConfig);
                });
